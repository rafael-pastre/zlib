284  uint8_t received_data[24]; // data received from BMP280 via i2c bus
285  HAL_StatusTypeDef status = HAL_OK; // status of the execution (checks if any error occurs)
286  /* Checks Sensor Initialization */

292  status = READ_REG(sensor->i2c_bus, BMP280_REG_STATUS, received_data);

295  return HAL_ERROR;

302  sensor->adc_T = ((uint32_t)received_data[3]<<12)+((uint32_t)received_data[4]<<4)+((uint32_t)received_data[5]>>4);
303 
304  /* Converts Temperature */
305  status = BMP280_compensate_temp(sensor);
306  RETURN_ON_ERROR(status);
307  sensor->temp = (float)sensor->raw_temp/100.0;

312  RETURN_ON_ERROR(status);
